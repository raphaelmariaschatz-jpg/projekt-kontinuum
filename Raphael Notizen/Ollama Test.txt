```python
class LowVolumeDiscoveryConnector:

        """Entdeckt und gibt eine Liste von globalen öffentlichen Wissens-APIs zurück.

        try:

            response.raise_for_status()
            data = response.json()

        except requests.RequestException as e:
            raise ValueError(f"Fehler beim Entdecken der APIs: {e}")
```

```python
def search_openalex(entity: str, query: str, limit: int) -> object:
    api_key = os.getenv("OPENALEX_API_KEY")
    if not api_key:
        raise RuntimeError("OPENALEX_API_KEY is required by OpenAlex.")

    # Sicherheitsprüfung der Query-Zeichenfolge
    safe_query = urllib.parse.quote(query)

    try:
        return get_json(
            f"https://api.openalex.org/{entity}",
            {"search": safe_query, "per-page": limit, "api_key": api_key},
        )
    except (RuntimeError, urllib.error.URLError, TimeoutError) as exc:
        print(json.dumps({"error": str(exc)}, ensure_ascii=False, indent=2), file=sys.stderr)
        raise
```
